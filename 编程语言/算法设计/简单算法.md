#include <stdio.h>
#include <string.h>
int main()
{
	char *sArray[5] ={"Mike","Andy","Tom","Jack","Mary"};
	char *temp;
	int i,j;
	
	for(i = 0;i<4;i++)
	{
		for(j = i+1;j<5;j++)
		{
			if(strcmp(sArray[i],sArray[j])>0)
			{
				temp = sArray[i];
				sArray[i] =sArray[j];
				sArray[j] = temp;
			}
		}
	}
	
	for(i = 0;i<5;i++)
		puts(sArray[i]);
	
	return 0;		
}
